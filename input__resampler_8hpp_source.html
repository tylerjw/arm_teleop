48 class InputResampler : public InputVisitor {
49  rclcpp::Node::SharedPtr node_ = nullptr;
50  rclcpp::Rate rate_;
51  std::shared_ptr<InputVisitor> next_ = nullptr;
52  mutable std::mutex mutex_;
53  InputCommand command_;
54  std::thread thread_;
55  std::atomic<bool> running_ = false;
56 
57  // Create a copy of the message so we don't have to lock durring the call to
58  // visit
59  InputCommand getCommandCopy() const {
60  const std::lock_guard<std::mutex> lock(mutex_);
61  return command_;
62  }
63 
64  // TODO(tylerjw): convert this back to a ROS timer after the performance of
65  // ros timers has been profiled
66  void threadMain() {
67  while (running_ and rclcpp::ok()) {
68  std::visit(*next_, getCommandCopy());
69  rate_.sleep();
70  }
71  }
72 
73  void startThread() {
74  running_ = true;
75  thread_ = std::thread([&]() { threadMain(); });
76  }
77 
78  void stopThread() {
79  if (running_) {
80  running_ = false;
81  thread_.join();
82  }
83  }
